use Exception;

class EmptyFieldException extends Exception
{
  /**
     * DBException constructor.
     * @param $mName
     */
    public function __construct($message, $code)
    {
        parent::__construct($message, $code);
    }


    public function __toString()
    {
        return $this->message;
    }
}
